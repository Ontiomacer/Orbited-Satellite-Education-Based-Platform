💡 USAGE EXAMPLES:

Example 1: Execute a Workflow
────────────────────────────────────────────────────────────────────
import { useWorqhat } from '@/hooks/useWorqhat';

function MyComponent() {
  const { runWorkflow } = useWorqhat();
  
  const execute = async () => {
    const result = await runWorkflow('your-workflow-id', {
      param1: 'value1'
    });
    console.log(result.data);
  };
}

Example 2: AI Chat
────────────────────────────────────────────────────────────────────
import { useWorqhat } from '@/hooks/useWorqhat';

function ChatComponent() {
  const { chat } = useWorqhat();
  
  const ask = async (question) => {
    const result = await chat(question);
    console.log(result.content);
  };
}

From useWorqhat() hook:
  • runWorkflow(workflowId, inputs) - Execute workflows
  • chat(question, options) - Get AI responses
  • checkStatus(executionId) - Check workflow status
  • loading - Boolean loading state
  • error - Error message if any
